-- HTTP request - WORKS ON ALL
function Universal.request(url)
    local success, result = pcall(function()
        if syn and syn.request then
            return syn.request({Url = url}).Body
        elseif request then
            return request({Url = url}).Body
        elseif http_request then
            return http_request({Url = url}).Body
        else
            return game:HttpGet(url)
        end
    end)
    return success and result or nil
end

-- Clipboard - WORKS ON ALL
function Universal.setClipboard(text)
    local success = pcall(function()
        if setclipboard then
            setclipboard(text)
            return true
        elseif toclipboard then
            toclipboard(text)
            return true
        elseif set_clipboard then
            set_clipboard(text)
            return true
        end
    end)
    return success or false
end

-- File system - WORKS IF SUPPORTED
function Universal.writeFile(path, content)
    pcall(function()
        if writefile then
            writefile(path, content)
        end
    end)
end

function Universal.readFile(path)
    local success, result = pcall(function()
        if isfile and readfile and isfile(path) then
            return readfile(path)
        end
        return nil
    end)
    return success and result or nil
end

function VelocitySystem:setupControls()
    -- Toggle with F key
    UserInputService.InputBegan:Connect(function(input)
        if input.KeyCode == Enum.KeyCode.F then
            if self.gui and self.gui.Parent then
                self.gui:Destroy()
                self.gui = nil
            else
                self:createGUI()
            end
        end
    end)
end

function VelocitySystem:startLoop()
    -- Main physics loop
    self.connections.loop = RunService.Heartbeat:Connect(function()
        if not self.enabled then return end
        if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then return end
        
        local root = player.Character.HumanoidRootPart
        local humanoid = player.Character:FindFirstChild("Humanoid")
        if not humanoid then return end
        
        -- Apply mode effects
        if self.mode == "Flight" then
            -- Flight mode
            root.Velocity = Vector3.new(root.Velocity.X, 0, root.Velocity.Z)
            
            -- Space to go up, Ctrl to go down
            if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
                root.Velocity = root.Velocity + Vector3.new(0, self.speed, 0)
            elseif UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
                root.Velocity = root.Velocity - Vector3.new(0, self.speed, 0)
            end
            
            -- Forward movement
            if UserInputService:IsKeyDown(Enum.KeyCode.W) then
                root.Velocity = root.Velocity + (workspace.CurrentCamera.CFrame.LookVector * self.speed)
            end
            if UserInputService:IsKeyDown(Enum.KeyCode.S) then
                root.Velocity = root.Velocity - (workspace.CurrentCamera.CFrame.LookVector * self.speed)
            end
            if UserInputService:IsKeyDown(Enum.KeyCode.A) then
                root.Velocity = root.Velocity - (workspace.CurrentCamera.CFrame.RightVector * self.speed)
            end
            if UserInputService:IsKeyDown(Enum.KeyCode.D) then
                root.Velocity = root.Velocity + (workspace.CurrentCamera.CFrame.RightVector * self.speed)
            end
            
        elseif self.mode == "Speed" then
            -- Speed mode - multiply velocity
            if UserInputService:IsKeyDown(Enum.KeyCode.W) or 
               UserInputService:IsKeyDown(Enum.KeyCode.A) or 
               UserInputService:IsKeyDown(Enum.KeyCode.S) or 
               UserInputService:IsKeyDown(Enum.KeyCode.D) then
                root.Velocity = root.Velocity * (1 + (self.speed / 100))
            end
            
        elseif self.mode == "Jump" then
            -- Super jump
            if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
                root.Velocity = Vector3.new(root.Velocity.X, self.speed * 2, root.Velocity.Z)
            end
            
        elseif self.mode == "NoClip" then
            -- No clip
            for _, part in ipairs(player.Character:GetChildren()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                end
            end
            
        elseif self.mode == "Gravity" then
            -- Zero gravity
            root.Velocity = root.Velocity * Vector3.new(1, 0.9, 1)
            humanoid.UseJumpPower = true
            humanoid.JumpPower = self.speed
        end
    end)
end

function ScriptHub:loadUniversalScripts()

    local scripts = {

            name = "Aimbot",

        },
        {
            name = "ESP",
            url = "https://raw.githubusercontent.com/Stefanuk12/ROBLOX/master/Universal/ESP.lua"
        },
        {
            name = "Fly",
            url = "https://raw.githubusercontent.com/Exunys/Airport/main/Airport.lua"
        }
    }
    
    for _, script in ipairs(scripts) do
        if not self.loaded[script.name] then
            pcall(function()
                local content = Universal.request(script.url)
                if content then
                    loadstring(content)()
                    self.loaded[script.name] = true
                    print("✅ Loaded: " .. script.name)
                end
            end)
        end
    end
end

    local submitCorner = Instance.new("UICorner")
    submitCorner.CornerRadius = UDim.new(0, 4)
    submitCorner.Parent = submit
    
    submit.MouseButton1Click:Connect(function()
        local key = input.Text
        if key and key ~= "" then
            self.key = key
            script_key = key
            Universal.writeFile("Pathos/key.txt", key)
            gui:Destroy()
            MainMenu:create()
        end
    end)
end
